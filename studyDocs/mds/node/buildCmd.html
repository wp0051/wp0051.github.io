自动化打包脚本案例

  • 执行命令 node buildCmd.js full-loan 或者 node buildCmd.js wap-api

目的

  • 读取项目配置：从项目目录下的 env.js 和 version.js 文件中提取环境名称和版本号。
  • 执行构建工具：运行 gulp (用于 assets 目录) 或 npm run build (用于 h5 目录) 来构建前端资源
  • 打包归档：使用 tar 命令将构建后的项目（并排除部分文件）打包成 .tar.gz 归档文件。
  • 清理：在打包前清理旧的归档文件目录

目录结构

├── buildCmd.js          # 下面脚本
├── full-loan/           # 'full-loan' 产品目录
│   ├── assets/          # 存放前端源码目录 (CSS, JS, 图片等)-jquery开发
│   │   ├── dist/        # (构建后) gulp 生成的最终静态资源文件
│   │   ├── rev-manifest.json # gulp 构建生成的资源映射文件 (根据脚本排除规则推测)
│   │   ├── node_modules
│   │   ├── gulpfile.js
│   │   └── package.json
│   ├── router/          # Node中间层路由--服务器渲染前端
│   ├── app.js           # Node启动文件
│   ├── env.js           # (脚本读取) 定义环境变量 (如 'dev', 'prod')
│   └── version.js       # (脚本读取) 定义版本号 (如 'v1.0.0')
└── wap-api/             # 'wap-api' 产品目录,项目结构与full-loan类似

脚本内容及解析

// 1. 引入 'child_process' 模块的 'exec' 函数，用于执行 shell 命令
// child_process 模块是 Node.js 用来创建和控制子进程的核心模块，其中 exec 函数是用于执行 shell 命令（包括脚本）的一个方法
var exec = require("child_process").exec;

// 2. 引入 'fs' 模块，用于文件系统操作
var fs = require("fs");

// 3. 引入 'path' 模块，用于处理文件路径
var path = require("path");

// 4. 定义一个异步函数 'runCmd'，用于执行 shell 命令并返回 Promise
const runCmd =async (cmd)=>{
    // 5. 返回一个新的 Promise，以便使用 async/await
    return new Promise((resolve, reject) => {
        // 6. 调用 child_process.exec 执行传入的命令 'cmd'
        //    第二个参数 {} 是选项对象，这里为空
        exec(cmd, {}, function (err, stdout, stderr) {
            // 7. 检查是否有错误 (exec 回调的第一个参数 err)
            if (err) {
                // 8. 如果有错误，拒绝 Promise 并传递错误对象
                reject( err);
            // 9. 检查 stderr 是否有内容
            } else if (stderr.length > 0) {
                // 10. 如果 stderr 有内容，拒绝 Promise 并创建一个包含 stderr 内容的错误
                reject( new Error(stderr.toString()));
            } else {
                // 11. 如果没有错误且 stderr 为空，解析 Promise 并传递命令的标准输出 stdout
                resolve(stdout);
            }
        })
    })
}

// 12. 定义一个异步函数 'readFile'，用于读取文件内容并返回 Promise
const readFile =   async(dir,filename)=>{
    // 13. 返回一个新的 Promise
    return new Promise((resolve,reject)=>{
        // 14. 使用 fs.readFile 读取文件
        //     路径是 dir 和 filename 的简单拼接 (不够安全，应使用 path.join)
        //     选项对象指定了文件标志 'r+' (读写，文件必须存在) 和编码 'utf8'
        fs.readFile(dir +filename, {flag: 'r+', encoding: 'utf8'}, function (err, data) {
            // 15. 检查读取文件时是否出错
            if(err) {
                // 16. 如果出错，拒绝 Promise 并传递错误对象
                reject(err);
            }else{
                // 17. 如果成功，解析 Promise 并传递读取到的文件内容 'data'
                resolve(data);
            }
        });
    })
}
// 18. 定义一个异步函数 'readDir'，用于读取目录内容并返回 Promise
const readDir = async (dirname)=>{
    // 19. 返回一个新的 Promise
    return new Promise((resolve, reject) => {
        // 20. 使用 fs.readdir 读取目录
        fs.readdir(dirname, function (err, files) {
            // 21. 检查读取目录时是否出错
            if (err) {
                // 22. 如果出错，拒绝 Promise 并传递错误对象
                reject(err);
            } else {
                // 23. 如果成功，解析 Promise 并传递文件/目录名数组 'files'
                resolve(files);
            }
        })
    })
}

// 24. 定义一个异步函数 'getProductNameAndVersion'，用于从项目文件中读取环境和版本
const getProductNameAndVersion =  async(product)=>{
    // 25. 使用 path.resolve 获取 'product' 目录的绝对路径
    let currentPath = path.resolve(__dirname,product);
    // 26. 定义一个正则表达式，用于匹配单引号包围的内容 (但在此处定义后未立即使用)
    let regex = /\'(.*?)\'/g;
    // 27. 读取 'currentPath' 目录下的 'env.js' 文件内容
    let env = await readFile(currentPath,"/env.js");
    // 28. 读取 'currentPath' 目录下的 'version.js' 文件内容
    let version = await readFile(currentPath,"/version.js");
    // 29. 如果 env 文件内容存在
    if(env){
        // 30. 重新定义正则表达式 (覆盖了上面的定义)
        let regex = /\'(.*?)\'/g;
        // 31. 使用正则表达式 exec 方法匹配 env 文件内容，获取第一个捕获组 (被单引号包围的字符串)
        var v= regex.exec(env)
        // 32. 将匹配到的内容 (v[1]) 赋值给 env 变量
        env = v[1];
    }
    // 33. 如果 version 文件内容存在
    if(version){
        // 34. 重新定义正则表达式 (覆盖了上面的定义)
        let regex = /\'(.*?)\'/g;
        // 35. 使用正则表达式 exec 方法匹配 version 文件内容，获取第一个捕获组
        var v= regex.exec(version)
        // 36. 将匹配到的内容 (v[1]) 赋值给 version 变量
        version = v[1];
    }
    // 37. 返回包含环境和版本的对象
    return {
        env,      // 环境名称
        version   // 版本号
    }
    
}   

// 38. 定义一个异步函数 'runCmdBuilding'，用于执行 gulp 构建命令
const runCmdBuilding = async (product)=>{
    // 39. 获取 'product' 目录的绝对路径
    let currentPath = path.resolve(__dirname,product);
    // 40. 构建 shell 命令：先切换到 'product/assets' 目录，然后执行 'gulp dist'
    let cmd = `cd ${currentPath}/assets && gulp dist`;
    // 41. 执行构建命令并返回其 Promise
    return await runCmd(cmd);
}

// 42. 定义一个异步函数 'runCmdH5'，用于执行 H5 (Vue) 项目的 npm 构建命令
const runCmdH5 = async (product)=>{
    // 43. 获取 'product' 目录的绝对路径
    let currentPath = path.resolve(__dirname,product);
    // 44. 构建 shell 命令：先切换到 'product/h5' 目录，然后执行 'npm run build'
    let cmd = `cd ${currentPath}/h5 && npm run build`;
    // 45. 执行 H5 构建命令并返回其 Promise
    return await runCmd(cmd);
}

// 46. 定义一个异步函数 'removeFlieAssest'，生成用于 tar 排除 assets 目录下非 dist 文件的参数
const removeFlieAssest = async (product)=>{
    // 47. 读取 './{product}/assets' 目录的内容
    let list = await readDir(`./${product}/assets`);
    // 48. 过滤掉 'dist' 和 'rev-manifest.json' 文件/目录
    list = list.filter((item)=>{
        return  item !="dist" && item !="rev-manifest.json"
    })
    // 49. 将过滤后的文件/目录名映射为 '--exclude=...' 格式的字符串
    list = list.map((item)=>{
        return  `--exclude=${product}/assets/${item}`
    })
    // 50. 将所有 '--exclude=...' 字符串用空格连接起来
    list = list.join(' ');
    // 51. 返回生成的排除参数字符串
    return  list;
}

// 52. 定义一个异步函数 'removeFlieH5'，生成用于 tar 排除 h5 目录下非 dist 文件的参数
const removeFlieH5 = async (product)=>{
    // 53. 读取 './{product}/h5' 目录的内容
    let list = await readDir(`./${product}/h5`);
    // 54. 过滤掉 'dist' 文件/目录
    list = list.filter((item)=>{
        return  item !="dist"
    })
    // 55. 将过滤后的文件/目录名映射为 '--exclude=...' 格式的字符串
    list = list.map((item)=>{
        return  `--exclude=${product}/h5/${item}`
    })
    // 56. 将所有 '--exclude=...' 字符串用空格连接起来
    list = list.join(' ');
    // 57. 返回生成的排除参数字符串
    return  list;
}

// 58. 定义一个异步函数 'runTar'，执行 tar 打包命令
const runTar = async (list,product,productname)=>{
    // 59. 构建 tar 命令
    //     - cd 切换到脚本所在目录
    //     - tar czf 创建压缩的 tar.gz 文件
    //     - 输出文件路径为 './{productname}/{product}.tar.gz'
    //     - --exclude 排除 .git 目录
    //     - ${list} 插入之前生成的排除参数
    //     - ${product} 指定要打包的目录
    const cmd  = `cd ${__dirname} && tar czf ./${productname}/${product}.tar.gz --exclude=${product}/.git ${list}  ${product}`
    // 60. 在控制台打印将要执行的命令
    console.log(cmd);
    // 61. 执行 tar 命令并返回其 Promise
    return await runCmd(cmd);
}

// 62. 定义一个异步函数 'runDelete'，用于删除指定的目录
const runDelete = async (product,productname)=>{
    // 63. 构建 rm 命令：切换到脚本目录并递归强制删除 '{productname}' 目录
    const cmd  = `cd ${__dirname} && rm -rf ${productname}`
    // 64. 执行删除命令并返回其 Promise
    return await runCmd(cmd);
}

// 65. 定义 'startFUllLOAN' 函数，用于执行 'full-loan' 产品的打包流程
const startFUllLOAN  =  async(product)=>{
    // 66. 调用函数获取产品的环境和版本信息
    const productinfo  = await getProductNameAndVersion(product);
    // 67. 根据产品名、环境和版本生成最终的打包目录名 (例如: full-loan_prod_v1.0)
    productname = `${product}_${productinfo.env}_${productinfo.version}`;
    // 68. 调用函数删除可能存在的旧打包目录
    await runDelete(product,productname);
    // 69. 构建创建新打包目录的命令
    var mkcmd = `mkdir ${productname}`;
    // 70. 执行创建目录命令
    await runCmd(mkcmd);
    // 71. 打印创建目录的命令
    console.log(mkcmd);
    // 72. 执行 assets 目录的 gulp 构建
    let result  = await runCmdBuilding(product)
    // 73. 打印 gulp 构建的输出 (如果有的话)
    console.log(result);
    // 74. 生成排除 assets 目录非 dist 文件的参数
    const list = await removeFlieAssest(product);
    // 75. 执行 tar 打包 (只包含 assets/dist)
    let result2 = await runTar(list,product,productname);
    // 76. 打印 tar 打包的输出 (如果有的话)
    console.log(result2);
}

// 77. 定义 'startWAPAPI' 函数，用于执行 'wap-api' 产品的打包流程
const startWAPAPI = async(product)=>{

    const productinfo  = await getProductNameAndVersion(product);
    // 79. 根据产品名、环境和版本生成最终的打包目录名
    productname = `${product}_${productinfo.env}_${productinfo.version}`;
    // 80. 调用函数删除可能存在的旧打包目录

    // 81. 构建创建新打包目录的命令
    var mkcmd = `mkdir ${productname}`;
    // 82. 执行创建目录命令
    await runCmd(mkcmd);
    // 83. 打印创建目录的命令
    console.log(mkcmd);
    // 84. 执行 assets 目录的 gulp 构建
    let result  = await runCmdBuilding(product)
    // 85. 打印 gulp 构建的输出 (如果有的话)
    console.log(result);
    // 86. 注释：执行 h5 打包
    // 87. 执行 h5 目录的 npm 构建
    let resulth5  = await runCmdH5(product)
    // 88. 打印 npm 构建的输出 (如果有的话)
    console.log(resulth5);
    // 89. 生成排除 assets 目录非 dist 文件的参数
    const list = await removeFlieAssest(product);
    // 90. 生成排除 h5 目录非 dist 文件的参数
    const list2 = await removeFlieH5(product);
    // 91. 注释：list (指上面生成的 list 和 list2)
    // 92. 执行 tar 打包 (包含 assets/dist 和 h5/dist)
    //     注意：这里使用了字符串拼接 'list+list2'，这会将两个字符串直接连接，中间没有空格。
    //     正确做法应该是 `list + ' ' + list2` 或者 `[list, list2].filter(Boolean).join(' ')`

    // 93. 打印 tar 打包的输出 (如果有的话)

}
