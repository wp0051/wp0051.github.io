File System

Node.js 的 fs（文件系统）模块 是用于操作文件和目录的核心模块，提供了同步和异步两种 API。

  • 异步 API (Async) 是 Node.js 推荐的主要使用方式。
  • 同步 API (Sync) 会阻塞事件循环，应谨慎使用，通常只在启动脚本或不频繁执行的场景中使用。
  • 基于 Promise 的 API (Promise) 提供了现代化的异步编程方式，易于与 async/await 结合使用。
  • 为了使用基于 Promise 的 API，需要 require('fs').promises 或 require('fs/promises')。
const fs = require('fs');
const fsPromises = require('fs').promises; // 或 require('fs/promises')

// const path = require('path');   Nodejs 的path模块

const filePath = "./example.txt";
// 也可以用path.join(__dirname, 'example.txt');__dirname:node全局变量，代表当前目录，这种写法兼容性好。
const dirPath = "./parent/child/nested"; 
// 也可以用 path.join，如 path.join(".", "parent", "child", "nested")

对文件进行创建、删除、写入、读取、重命名、存在判断操作

1.1 创建文件

1.1.1 writeFile

fs.writeFile(path, data[, options], callback) (异步)

// 异步创建文件并写入内容
fs.writeFile(filePath, 'Hello, async file!', (err) => {
  if (err) {
    console.error('异步创建/写入文件失败:', err.message);
    return;
  }
  console.log('异步文件创建/写入成功！');
});

fs.writeFileSync(path, data[, options]) (同步)

// 同步创建文件并写入内容

  console.log('同步文件创建/写入成功！');
} catch (err) {
  console.error('同步创建/写入文件失败:', err.message);
}

// 基于 Promise 创建文件并写入内容

  }

1.1.3 open

fs.open(path, flags[, mode], callback) (异步 - 用于获取文件描述符)

// 异步打开文件以获取文件描述符，'w' 模式会创建/覆盖文件
fs.open(filePath, 'w', (err, fd) => {
  if (err) {
    console.error('异步打开文件失败:', err.message);
    return;
  }
  console.log('异步打开文件成功，文件描述符:', fd);
  // ... 使用 fd 进行操作 ...
  fs.close(fd, (closeErr) => {
    if (closeErr) {
      console.error('关闭文件描述符失败:', closeErr.message);
    }
  });
});

fs.openSync(path, flags[, mode]) (同步)

// 同步打开文件以获取文件描述符
let fd;
try {
  fd = fs.openSync(filePath, 'w');
  console.log('同步打开文件成功，文件描述符:', fd);
  // ... 使用 fd 进行操作 ...
} catch (err) {
  console.error('同步打开文件失败:', err.message);
} finally {
  if (fd !== undefined) {
    fs.closeSync(fd); // 确保关闭
  }
}

fsPromises.open(path, flags[, mode]) (基于 Promise)

// 基于 Promise 打开文件以获取文件描述符
async function openFileWithPromise() {
  let fileHandle;
  try {
    fileHandle = await fsPromises.open(filePath, 'w');
    console.log('Promise 打开文件成功，文件句柄:', fileHandle.fd);
    // ... 使用 fileHandle 进行操作 ...
  } catch (err) {
    console.error('Promise 打开文件失败:', err.message);
  } finally {
    if (fileHandle) {
      await fileHandle.close(); // Promise 版本的 close
    }
  }
}

1.3.3 write

fs.write(fd, buffer, offset, length[, position], callback) (异步 - 使用文件描述符)

// 异步使用文件描述符写入
fs.open(filePath, 'a', (err, fd) => {
  if (err) {
    console.error('打开文件以写入失败:', err.message);
    return;
  }
  const buffer = Buffer.from('\nAsync write with fd.');
  fs.write(fd, buffer, 0, buffer.length, null, (writeErr, bytesWritten) => {
    if (writeErr) {
      console.error('异步写入失败:', writeErr.message);
    } else {
      console.log(`异步写入成功，字节数: ${bytesWritten}`);
    }
    fs.close(fd, (closeErr) => {
      if (closeErr) {
        console.error('关闭文件描述符失败:', closeErr.message);
      }
    });
  });
});

fs.writeSync(fd, buffer, offset, length[, position]) (同步 - 使用文件描述符)

// 同步使用文件描述符写入
let fd;
try {
  fd = fs.openSync(filePath, 'a');
  const buffer = Buffer.from('\nSync write with fd.');
  const bytesWritten = fs.writeSync(fd, buffer, 0, buffer.length, null);
  console.log(`同步写入成功，字节数: ${bytesWritten}`);
} catch (err) {
  console.error('同步写入失败:', err.message);
} finally {
  if (fd !== undefined) {
    fs.closeSync(fd);
  }
}

fileHandle.write(buffer[, options]) (基于 Promise - 使用文件句柄)

// 基于 Promise 使用文件句柄写入
async function writeWithHandlePromise() {
  let fileHandle;
  try {
    fileHandle = await fsPromises.open(filePath, 'a');
    const buffer = Buffer.from('\nPromise write with handle.');
    const result = await fileHandle.write(buffer);
    console.log(`Promise 写入成功，字节数: ${result.bytesWritten}`);
  } catch (err) {
    console.error('Promise 写入失败:', err.message);
  } finally {
    if (fileHandle) {
      await fileHandle.close();
    }
  }
}
// 调用示例: writeWithHandlePromise();

2.1 创建文件夹

2.1.1 mkdir

fs.mkdir(path[, options], callback) (异步)

// 异步创建单个目录
fs.mkdir(dirPath, (err) => {
  if (err) {
    if (err.code === 'EEXIST') {
      console.log('异步创建: 目录已存在。');
    } else {
      console.error('异步创建目录失败:', err.message);
    }
    return;
  }
  console.log('异步目录创建成功！');
});
// 异步递归创建目录{ recursive: true }
fs.mkdir(dirPath, { recursive: true }, (err) => {
  if (err) {
    console.error('异步递归创建目录失败:', err.message);
    return;
  }
  console.log('异步递归目录创建成功！路径:', dirPath);
});

3.2 重命名及移动文件/文件夹

3.2.1 rename

fs.rename(oldPath, newPath, callback) (异步) rename接收的完整的路径，不是一个文件或者文件夹名，所以可以做到移动文件/文件夹

// 异步重命名文件/文件夹
const oldFile = path.join(__dirname, oldfileName);
const newFile = path.join(__dirname, newfileName);
// 假设 oldFile 存在
fs.writeFile(oldFile, 'Temp content', (err) => {
  if (err) throw err;
  fs.rename(oldFile, newFile, (err) => {
    if (err) {
      console.error('异步重命名文件失败:', err.message);
      return;
    }
    console.log(`异步文件已重命名从 ${oldFile} 到 ${newFile}，也可以说从 ${oldFile} 移动到 ${newFile}`);
  });
});

// 异步重命名目录
const oldDir = path.join(__dirname, oldDirName);
const newDir = path.join(__dirname, newDirName);
// 假设 oldDir 存在
fs.mkdir(oldDir, (err) => {
  if (err && err.code !== 'EEXIST') throw err;
  fs.rename(oldDir, newDir, (err) => {
    if (err) {
      console.error('异步重命名目录失败:', err.message);
      return;
    }
    console.log(`异步目录已重命名从 ${oldDir} 到 ${newDir}，也可以说从 ${oldDir} 移动到 ${newDir}`);
  });
});

fs.renameSync(oldPath, newPath) (同步)
