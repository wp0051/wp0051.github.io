3. nodeJs的文件系统模块-fs

1. 读取文件

例如，可以利用fs读取一个txt文本文件：

const fs = require('fs')

fs.readFile('./index.txt', (err, data) => {
  if (err) throw err
  console.log(data.toString()) // 将读取的二进制数据通过toString转换成可读字符串
})

也可通过给readFile方法，传入options配置，直接输出字符串数据

// 读取文件
fs.readFile('./index.txt', { encoding: 'utf8' }, (err, data) => {
  if (err) throw err
  console.log(data)
})

2. 写入文件

如果要写入的文件，不存在文件夹内，则会先创建这个文件，并写入数据：

// 写入文件
fs.writeFile('./index2.txt', 'hello world', err => {
  if (err) throw err
  console.log('写入成功')
})

3. 追加写入文件

writeFile方法写入是整体覆盖，如果需要追加写入，你需要这样操作：

// 追加写入文件
const buf = Buffer.alloc(16, '世界和平')
fs.appendFile('./index.txt', buf, err => {
  if (err) throw err
  console.log('追加成功')
})

4. 获取文件信息

主要用来判断是文件还是文件夹

// 获取文件信息
fs.stat('./index.txt', (err, stats) => {
  if (err) throw err
  console.log(stats)
  console.log(stats.isFile())
  console.log(stats.isDirectory())
})

5. 重命名文件

// 重命名文件
fs.rename('./index.txt', './test.txt', err => {
  if (err) throw err
})

6. 删除文件

// 删除文件
fs.unlink('./index2.txt', err => {
  if (err) throw err
})

7. 创建文件夹

// 创建文件夹
fs.mkdir('./test',  err => {
  if (err) throw err
})

也可以递归创建文件夹：

// 递归创建文件夹
fs.mkdir('./test/test1/test1-1', {
  recursive: true
}, err => {
  if (err) throw err
})

8. 读取文件夹

// 读取文件夹
fs.readdir('./test', {
  withFileTypes: true // 返回时，把文件类型加上
}, (err, files) => {
  if (err) throw err
  console.log(files)
})

9. 删除文件夹

// 删除文件夹
// 会删除test1-1文件夹
fs.rmdir('./test/test1/test1-1', err => {
  if (err) throw err
})

递归删除文件夹

// 递归删除文件夹
// test文件夹下的所有文件夹会逐步删除
fs.rm('./test', {
  recursive: true
}, err => {
  if (err) throw err
})

10. 监听文件的变化

fs.watch('./', {
  recursive: true // 深度监听，子目录的变化也会被监听到
}, (eventType, fileName) => {
  console.log(eventType, fileName)
})

利用watch监听的时候，一个文件的变化导致2次监听结果输出，所以如果需要监听文件的改变，可以使用chokidar这个插件：

npm/cnpm install chokidar --save-dev

使用这个插件：

const chokidar = require('chokidar')
chokidar.watch('./', {
    ignored: './node_modules', // 不监听node_modules
}).on('all', (event, files) => {
    console.log(event, files);
})
